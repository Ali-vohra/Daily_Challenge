from collections import defaultdict

grid = ["2 8 8 6 4", "8 2 4 2 12", "4 4 2 6 16", "6 8 4 12 2", "8 6 4 2 4"]
ROWS = len(grid) # 5
COLUMNS = len(grid[0].split()) # 5
solutions = defaultdict(list)

# checks all possible zeros in each row resulting 16
def rowCheck(rowSolution, zeros, gridRow, used, gridSaved):
    global ROWS, COLUMNS
    if zeros == 0:
        if sum(list(map(int, gridRow))) == 16:
            if gridRow not in solutions[rowSolution]:
                solutions[rowSolution].append(gridRow.copy())
        return
    else:
        for g in range(ROWS):
            if g in used:
                continue
            else:
                gridRow[g] = "0"
                used.add(g)
                rowCheck(rowSolution, zeros - 1, gridRow, used, gridSaved)
                used.remove(g)
                gridRow[g] = gridSaved[g]

# checks the sum of columns of all grid-combinations with rows resulting 16
def columnCheck(n, gridSolution, used):
    global ROWS, COLUMNS
    if n == COLUMNS:
        counter = 0
        for column in range(COLUMNS):
            amount  = 0
            for row in range(ROWS):
                    amount += int(gridSolution[row][column])
            if amount == 16:
               counter += 1
        if counter == COLUMNS:
            for g in gridSolution:
                print(g)
            print(sum(g.count("0") for g in gridSolution), "zeros")
        return
    else:
        for solution in solutions[n]:
            if tuple(solution) in used:
                continue
            else:
                gridSolution[n] = solution
                used.add(tuple(solution))
                columnCheck(n + 1, gridSolution, used)
                used.remove(tuple(solution))

#  main
for i, gridRow in enumerate(grid):
    for zeros in range(4, 0, -1):
        rowCheck(i, zeros, gridRow.split(), set(), gridRow.split())
columnCheck(0, ["", "", "", "", ""], set())

In order to have a sum of sixteen in that column, we need to get rid of a total of 6. This can be done with a combination of either 4's and the 2. Therefore, the 2 must be 0.

6, 4 and 4 add up to 14, which isn't enough. So we need to have the 16 only.

We have to keep the 16 here. So every other number in that column must be 0.

There is no way to add up to 16 with only 6, 8 and 4. So we need to include the 12 and, necessarily, the 4. 6 and 8 are turned to zeros.

The bottom 4 is the only one that can be dispensed. Same for the top-left 2. Then the top 6 obviously have to be a zero. And we're done!

That's a total of 13 zeros.

There is no need for trial and error in this case. Just apply logic similar to that for a sudoku puzzle.

Start with the center column. The 2 cannot contribute to 16, so it is a zero. That means that the 16 must be used for the center row. So the 4, 4, and 6 are also zeroes

Moving to the far right column, all of the numbers except the 16 must be zeroes.

Moving to the second column from the right, the top 6 must also be zero, leaving the 12, 2, and 2 to make 16.

Moving to the far left of the top row, the The 2 must be zero, leaving the two 8â€™s to make 16.

Now the far left column. With the 2 out of the picture, the only way to make 16 is 8+8. So the 6 must also be zero.

Drop down to the second row from the bottom. The only way to get 16 is 4 and 12, so the 8 becomes zero.

On the bottom row, the center 4 must also be zero, leaving 8, 6, and 2 for 16.

No guessing required.
