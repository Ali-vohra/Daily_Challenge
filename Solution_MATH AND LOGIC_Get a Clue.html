class Block:
    def __init__(self, characters, directions):
        # container class; directions are tuples indicating which direction to move to place the next character
        self.chars = characters
        directions.append((0, 0)) # to make sure directions and characters have the same len
        self.dirs = directions

def place(board, x, y, block):
    for i in range(0, len(block.chars)):
        if 0 <= x < len(board) and 0 <= y < len(board[x]) and board[x][y] == ' ':
            # We're allowed to place the character here
            board[x][y] = block.chars[i]
            x, y = x + block.dirs[i][0], y + block.dirs[i][1] # Moving the coordinates for the next step
        else:
            # Block can't fit here, oh no! We have to undo the progress we've made
            for j in range(i-1, -1, -1):
                # Move backwards, undoing every move that we made
                x, y = x - block.dirs[j][0], y - block.dirs[j][1]
                board[x][y] = ' '
            return False # Indicating an unsuccessful placement
    return True # Indicating a successful move


def undo_placement(board, x, y, block):
    for j in range(0, len(block.dirs)):
        board[x][y] = ' '
        x, y = x + block.dirs[j][0], y + block.dirs[j][1]

def place_blocks(index, blockslist, board):
    if index == len(blockslist):
        # We've made it to the end of the list, 
        # having successfully placed every block onto the grid
        print("Solution:")
        for i in range(0, len(board)):
            for j in range(0, len(board[i])):
                print(board[i][j], end=' ')
            print()
        print()

    else:
        # Try to place the current block in every possible position;
        # if it succeeds, place the next block
        for i in range(0, len(board)):
            for j in range(0, len(board[i])):
                successful_place = place(board, i, j, blockslist[index])
                if successful_place:
                    place_blocks(index + 1, blockslist, board)
                    undo_placement(board, i, j, blockslist[index])

if __name__ == "__main__":
    # note: (-1, 0) means "move up," (1, 0) means "move down", (0, 1) means "move right," (0, -1) means "move left"
    s_piece = Block(['h', 'f', 'g', 'e'], [(-1,0), (0, 1), (-1, 0)])
    l_piece = Block(['z', 'y', 'w', 'x'], [(-1, 0), (-1, 0), (0, 1)])
    c_piece = Block(['j', 'l', 'm', 'n', 'k'], [(1, 0), (0, 1), (0, 1), (-1, 0)])
    two_piece = Block(['q', 'r'], [(0, 1)])

    big_board = []
    for i in range(0, 4):
        big_board.append([' ', ' ', ' ', ' ']) # defining the board outlined in the problem, simple 4x4 grid

    place_blocks(0, [s_piece, l_piece, c_piece, two_piece], big_board)
