    from itertools import permutations

solutions = []

def proof(n, chest, permutationTrueFalse, used):
    if n == 4: # permutation of  1 false and 3 true should be equal to
               # evaluation of statements
        if permutationTrueFalse[chest.index("gold")] == True:
            if [not chest[0] in ["gold", "mix"],
                not chest[1] in ["gold", "mix"],
                    chest[2] in ["mix"],
                    chest[0] in ["mix"]] == permutationTrueFalse:
                if (chest.copy(), permutationTrueFalse) not in solutions:
                    solutions.append((chest.copy(), permutationTrueFalse))
        return
    else:
        for item in ["gold", "silver", "mix", "rock"]:
            if item in used:
                continue
            chest[n] = item
            used.append(item)
            proof(n+1, chest, permutationTrueFalse, used)
            used.remove(item)

for p in permutations([False, True, True, True]):
    proof(0, ["", "", "", ""], list(p), [])
for s in solutions:
  print(s)
print(f"{len(solutions)} solutions")
